        return 1;
    }
    if (argc == 3 && strcmp(argv[2], "--sleep") == 0)
    {
        sleep_enabled = true;
    }
    char *filename = argv[1];
    pid_t pid = fork(); // Get the current process ID
    if (pid < 0)
    {
        perror("Failed to fork");
        return 1;
    }
    else if  (pid == 0)
    {
        // Child process
        do_the_same_thing(filename, sleep_enabled);
        exit(0);
    }
    else if (pid > 0)
    {
        do_the_same_thing(filename, sleep_enabled);
        
    }
    else
    {
        perror("Failed to fork");
        return 1;
    }

    return 0;
}
