      $$
      0\leq r(A)\leq \min\{m,n\}
      $$

      $$
      r(kA) = r(A)(k\neq0)
      $$

      $$
      r(AB)\leq\min\{r(A),r(B)\}
      $$

      $$
      r(A+B)\leq r(A) + r(B)
      $$

      $$
      r(A)=\begin{cases}
      n,r(A)=n\\
      1,r(A)=n-1\\
      0,r(A)<n-1
      
      \end{cases}
      $$

      $$
      r(A) = r(PA) = r(AQ) = r(PAQ)
      $$

      这里$P$，$Q$为可逆矩阵
      $$
      A_{m\times n}B_{n\times s} = O \Rightarrow r(A) + r(B) \leq n
      $$

      $$
      r(A) = r(A^T) = r(AA^T)=r(A^TA)
      $$
